目的：
輸出「熱門收物帖」清單。根據近期瀏覽數、回覆數與新貼加成計分，加入輕微抖動打散同分；（非 owner 模式）再做同作者上限與 Softmax 機率抽樣（含探索比例），最後寫入推薦歷史以支援首頁冷卻。

作法

候選集：
owner 模式：只調整候選（自己看含私密，他人僅公開），不套 NEW_DAYS 限制。
一般模式：公開且 NEW_DAYS 內有更新；排除黑名單與自己。
首頁可套 cooldown_days（支援登入/匿名 session）；候選過少時有分層回退。


打分：按視窗內（days）瀏覽數 × w、回覆數 × w，再加上新貼加成。


排序/抽樣：
owner 模式：只按分數排序，不做多樣性與抽樣。
一般模式：同作者最多 K=5 → 取池（最多 10×limit）→ Softmax(T=0.7) 抽樣，混入 explore_ratio。


輸出/記錄：保序回傳；一般模式寫入 WantRecommendationHistory(source='hot_rank', algorithm_version='v3')，owner 模式不寫歷史以免影響首頁冷卻。


# 候選與冷卻
now = timezone.now()
recent = now - timedelta(days=days)
user = getattr(request, 'user', None)
is_auth = bool(user and getattr(user, "is_authenticated", False))
blocked_ids = set(get_blocked_user_ids(user)) if is_auth else set()
owner_mode = owner is not None
is_homefeed = (not owner_mode) and (not keyword) and (not tag)  # 只首頁流寫歷史/套冷卻


# 候選
if owner_mode:
    qs = Want.objects.filter(user=owner, permission__id__in=([1,2] if (user and owner==user) else [1]))
    qs = qs.exclude(user_id__in=blocked_ids).distinct()
else:
    qs = (Want.objects.filter(permission__id=1)
          .exclude(user_id__in=blocked_ids).exclude(user=user)
          .filter(update__gte=now - timedelta(days=NEW_DAYS)))
    if keyword:
        qs = qs.filter(Q(title__icontains=keyword) | Q(post_text__icontains=keyword) |
                       Q(wanttag__tag__name__icontains=keyword) | Q(user__username__icontains=keyword))
    if tag:
        qs = qs.filter(wanttag__tag=tag)
    qs = qs.distinct()

# 首頁冷卻（登入/匿名皆可）
is_homefeed = not (owner or keyword or tag)
if is_homefeed and cooldown_days > 0:
    ids_to_exclude = fetch_recent_want_reco_ids_for_user_or_session(request, cooldown_days)  # 你的現有流程
    remain = qs.exclude(id__in=ids_to_exclude)
    qs = fallback_if_too_strict(qs, remain)  # 12 小時 / 30% / 還原 的分層回退

#熱度打分
scores = {wid: 0.0 for wid in qs.values_list('id', flat=True)}

# 瀏覽
for row in (WantFootprints.objects
            .filter(date__gte=recent, want_id__in=scores.keys())
            .values('want_id').annotate(c=Count('id'))):
    scores[row['want_id']] += HOT_WEIGHTS.get('recent_views', 0) * row['c']

# 回覆
for row in (WantBack.objects
            .filter(date__gte=recent, want_id__in=scores.keys())
            .values('want_id').annotate(c=Count('id'))):
    scores[row['want_id']] += HOT_WEIGHTS.get('recent_replies', 0) * row['c']

# 新貼加成
if HOT_WEIGHTS.get('new_want_bonus', 0):
    for wid in Want.objects.filter(id__in=scores.keys(),
                                   update__gte=now - timedelta(days=NEW_DAYS)
                                   ).values_list('id', flat=True):
        scores[wid] += HOT_WEIGHTS['new_want_bonus']

# 輕微抖動（依 user/session + 時間種子）
seed_id = f"user-{user.id}" if (user and user.is_authenticated) else f"sess-{getattr(request.session, 'session_key', 'anon')}"
time_key = now.strftime('%Y%m%d%H' if seed_scope=='hour' else '%Y%m%d')
rnd = random.Random(f"want-hot-jitter|{seed_id}|{time_key}|o={getattr(owner,'id',owner)}|k={bool(keyword)}|t={getattr(tag,'id',tag)}")
for wid in scores:
    scores[wid] += rnd.uniform(-jitter, jitter)

# 多樣性與抽樣
prelim = sorted(scores.keys(), key=lambda wid: scores[wid], reverse=True)

if owner_mode:
    final_ids = prelim[: (limit or len(prelim))]
else:
    # 同作者最多 K=5
    author_by = dict(Want.objects.filter(id__in=prelim).values_list('id', 'user_id'))
    cap, cnt, pool = 5, defaultdict(int), []
    for wid in prelim:
        aid = author_by.get(wid)
        if cnt[aid] < cap:
            pool.append(wid); cnt[aid] += 1
        if limit and len(pool) >= (limit * 10):
            break

    # Softmax 抽樣 + 探索
    def softmax(vals, T=0.7):
        if not vals: return []
        vmax = max(vals); exps = [math.exp((v - vmax)/T) for v in vals]
        s = sum(exps) or 1.0; return [x/s for x in exps]

    probs = softmax([scores[wid] for wid in pool])
    if 0 < explore_ratio < 1:
        uni = 1.0/len(pool); probs = [(1-explore_ratio)*p + explore_ratio*uni for p in probs]

    rng = random.Random(f"want-hot-pick|{seed_id}|{time_key}|o={getattr(owner,'id',owner)}|k={bool(keyword)}|t={getattr(tag,'id',tag)}")
    need, picked, cand, w = (limit or 20), [], pool[:], probs[:]
    for _ in range(min(need, len(cand))):
        tot = sum(w); idx = rng.randrange(len(cand)) if tot <= 0 else next(
            i for i, acc in enumerate([sum(w[:i+1]) for i in range(len(w))]) if rng.uniform(0, tot) <= acc
        )
        picked.append(cand.pop(idx)); w.pop(idx)
    final_ids = picked if picked else prelim[: (limit or 20)]

# 保序與寫歷史
preserved = Case(*[When(id=pk, then=pos) for pos, pk in enumerate(final_ids)], output_field=IntegerField())
qs_ordered = Want.objects.filter(id__in=final_ids).order_by(preserved)

if not owner_mode:
    now_ts = timezone.now()
    WantRecommendationHistory.objects.bulk_create(
        [
            WantRecommendationHistory(
                user=user, want=w, recommended_at=now_ts,
                source='hot_rank', algorithm_version='v3',
                keyword=keyword or None
            )
            for w in qs_ordered
        ],
        ignore_conflicts=True,
    )

return qs_ordered
