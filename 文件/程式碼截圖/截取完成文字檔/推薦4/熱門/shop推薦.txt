目的：
產生「熱門商店」清單。依近窗瀏覽數、成交數與新店加成計分，加入輕微抖動打散同分；（非 owner 模式）再做同賣家上限與 Softmax 機率抽樣（含探索比例），最後寫入推薦歷史以支援首頁冷卻。

作法
候選集：
owner 模式：只調整候選（自己看含私密 / 看別人僅公開），不套 NEW_DAYS/進行中限制。
一般模式：僅公開店，且進行中或 NEW_DAYS 內更新；排除黑名單與自己。
首頁可套 cooldown_days：排除使用者（或 session）近 N 天已推的店，候選過少時有分層回退。

打分：依視窗內（days）的瀏覽數與成交數乘權重，再加上新店加成（NEW_DAYS 內）。

排序/多樣性/抽樣：
owner 模式：只按分數排序，不做多樣性與抽樣。
一般模式：同 owner 最多 K=5 家 → 取池（最多 10×limit）→ Softmax(T=0.7) 抽樣，混入 explore_ratio。

輸出/記錄：保序回傳；一般模式寫入 ShopRecommendationHistory(source='hot_rank', algorithm_version='v3')保存推薦記錄


# 候選與冷卻
now = timezone.now()
recent = now - timedelta(days=days)
user = getattr(request, 'user', None)
blocked_ids = set(get_blocked_user_ids(user)) if (user and user.is_authenticated) else set()

owner_mode = owner is not None
if owner_mode:
    qs = Shop.objects.filter(owner=owner, permission_id__in=([1,2] if (user and owner==user) else [1]))
    qs = qs.exclude(owner_id__in=blocked_ids).distinct()
else:
    qs = (Shop.objects.filter(permission_id=1)
          .exclude(owner_id__in=blocked_ids).exclude(owner=user)
          .filter(Q(start_time__lte=now, end_time__gte=now) | Q(update__gte=now - timedelta(days=NEW_DAYS))))
    if keyword:
        qs = qs.filter(Q(name__icontains=keyword) | Q(introduce__icontains=keyword) |
                       Q(shoptag__tag__name__icontains=keyword) | Q(owner__username__icontains=keyword))
    if tag:
        qs = qs.filter(shoptag__tag=tag)
    qs = qs.distinct()

    # 首頁冷卻（含匿名 session）
    is_homefeed = not (keyword or tag)
    if is_homefeed and cooldown_days > 0:
        ids_to_exclude = fetch_recent_reco_ids_for_user_or_session(request, cooldown_days)  # 你的現有邏輯
        remain = qs.exclude(id__in=ids_to_exclude)
        qs = fallback_if_too_strict(qs, remain)  # 你的分層回退：12 小時 / 30% / 還原

# 熱度打分
scores = {sid: 0.0 for sid in qs.values_list('id', flat=True)}

# 瀏覽
for row in (ShopFootprints.objects
            .filter(date__gte=recent, shop_id__in=scores.keys())
            .values('shop_id').annotate(c=Count('id'))):
    sid = row['shop_id']
    scores[sid] += HOT_WEIGHTS['recent_views'] * row['c']

# 成交（有效訂單：state 4、5）
for row in (ProductOrder.objects
            .filter(order__date__gte=recent,
                    order__order_state__id__in=[4,5],
                    product__shop_id__in=scores.keys())
            .values('product__shop_id').annotate(c=Count('id'))):
    sid = row['product__shop_id']
    scores[sid] += HOT_WEIGHTS['recent_sales'] * row['c']

# 新店加成
for sid in Shop.objects.filter(id__in=scores.keys(),
                               update__gte=now - timedelta(days=NEW_DAYS)
                               ).values_list('id', flat=True):
    scores[sid] += HOT_WEIGHTS['new_shop_bonus']

# 輕微抖動（依 user/session + 時間種子）
seed_id = f"user-{user.id}" if (user and user.is_authenticated) else f"sess-{getattr(request.session, 'session_key', 'anon')}"
time_key = now.strftime('%Y%m%d%H' if seed_scope=='hour' else '%Y%m%d')
rnd = random.Random(f"hot-jitter|{seed_id}|{time_key}|o={getattr(owner,'id',owner)}|k={bool(keyword)}|t={getattr(tag,'id',tag)}")
for sid in scores:
    scores[sid] += rnd.uniform(-jitter, jitter)

# 多樣性與抽樣
prelim = sorted(scores.keys(), key=lambda sid: scores[sid], reverse=True)

if owner_mode:
    final_ids = prelim[: (limit or len(prelim))]
else:
    # 同 owner 最多 K=5
    owner_by = dict(Shop.objects.filter(id__in=prelim).values_list('id', 'owner_id'))
    cap, cnt, pool = 5, defaultdict(int), []
    for sid in prelim:
        oid = owner_by.get(sid)
        if cnt[oid] < cap:
            pool.append(sid); cnt[oid] += 1
        if limit and len(pool) >= (limit * 10):
            break

    # Softmax 抽樣 + 探索
    def softmax(vals, T=0.7):
        if not vals: return []
        vmax = max(vals); exps = [math.exp((v - vmax)/T) for v in vals]
        s = sum(exps) or 1.0; return [x/s for x in exps]

    probs = softmax([scores[sid] for sid in pool])
    if 0 < explore_ratio < 1:
        uni = 1.0/len(pool); probs = [(1-explore_ratio)*p + explore_ratio*uni for p in probs]

    rng = random.Random(f"hot-pick|{seed_id}|{time_key}|o=None|k={bool(keyword)}|t={getattr(tag,'id',tag)}")
    need, picked, cand, w = (limit or 20), [], pool[:], probs[:]
    for _ in range(min(need, len(cand))):
        tot = sum(w); idx = rng.randrange(len(cand)) if tot <= 0 else next(
            i for i, acc in enumerate([sum(w[:i+1]) for i in range(len(w))]) if rng.uniform(0, tot) <= acc
        )
        picked.append(cand.pop(idx)); w.pop(idx)
    final_ids = picked if picked else prelim[: (limit or 20)]

# 保序與寫歷史
preserved = Case(*[When(id=pk, then=pos) for pos, pk in enumerate(final_ids)], output_field=IntegerField())
qs_ordered = Shop.objects.filter(id__in=final_ids).order_by(preserved)

if not owner_mode:
    now_ts = timezone.now()
    payloads = []
    for s in qs_ordered:
        payload = dict(shop=s, recommended_at=now_ts, source='hot_rank', algorithm_version='v3')
        if keyword: payload['keyword'] = keyword
        if user and user.is_authenticated:
            payload['user'] = user
        else:
            if request and not request.session.session_key: request.session.save()
            sess_key = getattr(request.session, 'session_key', None)
            if not sess_key: continue
            payload['session_key'] = sess_key
        payloads.append(ShopRecommendationHistory(**payload))
    if payloads:
        ShopRecommendationHistory.objects.bulk_create(payloads, ignore_conflicts=True)

return qs_ordered
